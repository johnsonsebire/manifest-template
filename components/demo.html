Usage

<!-- Include in your HTML --> <div id="mega-menu-placeholder"></div> <div id="dropdown-placeholder"></div> <script> // Load components async function loadComponent(componentName, targetId) { try { const response = await fetch(`components/${componentName}.html`); const html = await response.text(); document.getElementById(targetId).innerHTML = html; } catch (error) { console.error(`Failed to load ${componentName}:`, error); } } // Initialize after DOM loads document.addEventListener('DOMContentLoaded', async function() { await Promise.all([ loadComponent('mega-menu', 'mega-menu-placeholder'), loadComponent('drop-down', 'dropdown-placeholder') ]); }); </script>
